# 279. Perfect Squares
  

### Solution 1 - Beat 27.62%

1. make dp table which contains minimum number of prime numbers which sum up to i.

2. dp[t] = dp[closet prime numer to t(=x)] + dp[x] 

```cpp
class Solution {
public:
    bool isprime(int n){
        for(int i=1; i<n; i++){
            if(i*i == n)
                return true;
            if(i*i > n)
                return false;
        }
        return false;
    }
    int numSquares(int n) {
        vector<int> dp;
        if(n<=1)
            return n;
        dp.push_back(1); dp.push_back(2);
        vector<int> prime; prime.push_back(1);
        for(int i=3; i<=n; i++){
            if(isprime(i)){
                dp.push_back(1);
                prime.push_back(i);
            }else{
                int tmp = i;
                for(auto p : prime){
                    tmp = min(tmp, 1+dp[(i-1) - (p)] );
                }
                dp.push_back(tmp);
            }
        }
        
        return dp[n-1];
    }
};

```
