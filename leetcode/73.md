#include <set>

class Solution {
public:
    void setZeroes(vector<vector<int>>& matrix) {
        set<int> rs,cs;
        int row=0, col = 0;
        
        row = matrix.size();
        if(row == 0 ) return;
        col = matrix[0].size();
        if(col == 0) return;
        
        for(int i=0; i<row; i++){
            for(int j=0; j<col; j++){
                if(matrix[i][j]==0){
                    rs.insert(i);
                    cs.insert(j);
                }
            }
        }
        
        // make every row in rs to zeros
        for (set<int>::iterator itr = rs.begin(); itr != rs.end(); ++itr) {
            for(int i=0; i<col; i++){
                matrix[*itr][i] =0;
            }
        }
        
        // make every col in cs to zeros
        for (set<int>::iterator itr = cs.begin(); itr != cs.end(); ++itr) {
            for(int i=0; i<row; i++){
                matrix[i][*itr] =0;
            }
        }
        
        
    }
};
